⚡ 快速开始 - Python 代码示例

安装依赖：pip install openai

# OpenRouter 示例 (推荐) from openai import OpenAI client = OpenAI( api_key="YOUR_OPENROUTER_KEY", base_url="https://openrouter.ai/v1" ) response = client.chat.completions.create( model="meta-llama/llama-3.2-3b-instruct:free", messages=[{"role": "user", "content": "你好!"}] ) print(response.choices[0].message.content)
# Google AI Studio 示例 import google.generativeai as genai genai.configure(api_key="YOUR_GEMINI_KEY") model = genai.GenerativeModel('gemini-2.0-flash') response = model.generate_content("Hello!") print(response.text)
# Groq 示例 (极速推理) from openai import OpenAI client = OpenAI( api_key="YOUR_GROQ_KEY", base_url="https://api.groq.com/openai/v1" ) response = client.chat.completions.create( model="llama-3.1-70b-versatile", messages=[{"role": "user", "content": "你好!"}] )
